server {
    server_name 72.61.125.165 maryamhostel.com www.maryamhostel.com;

# Proxy /admin to Django backend
location /admin/ {
    proxy_pass http://127.0.0.1:8001/admin/;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
}

    # Proxy all /api/ requests to Django (Gunicorn)
    location /api/ {
        proxy_pass http://127.0.0.1:8001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Serve React frontend
    root /var/www/maryamhostel/frontend;
    index index.html;

    location / {
        try_files $uri /index.html;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/maryamhostel.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/maryamhostel.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}

server {
    if ($host = www.maryamhostel.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = maryamhostel.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name 72.61.125.165 maryamhostel.com www.maryamhostel.com;
    return 404; # managed by Certbot




}
